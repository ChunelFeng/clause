116 void Trainer::append(const ItemSequence& xseq, const StringList& yseq, int group) {
117  // Create dictionary objects if necessary.
118  if (data->attrs == NULL || data->labels == NULL) {
119  init();
120  }
121 
122  // Make sure |y| == |x|.
123  if (xseq.size() != yseq.size()) {
124  std::stringstream ss;
125  ss << "The numbers of items and labels differ: |x| = " << xseq.size() << ", |y| = " << yseq.size();
126  throw std::invalid_argument(ss.str());
127  }
128 
129  // Convert instance_type to crfsuite_instance_t.
130  crfsuite_instance_t _inst;
131  crfsuite_instance_init_n(&_inst, xseq.size());
132 
133  for (size_t t = 0; t < xseq.size(); ++t) {
134  const Item& item = xseq[t];
135  crfsuite_item_t* _item = &_inst.items[t];
136 
137  // Set the attributes in the item.
138  crfsuite_item_init_n(_item, item.size());
139 
140  for (size_t i = 0; i < item.size(); ++i) {
141  _item->contents[i].aid = data->attrs->get(data->attrs, item[i].attr.c_str());
142  _item->contents[i].value = (floatval_t)item[i].value;
143  }
144 
145  // Set the label of the item.
146  _inst.labels[t] = data->labels->get(data->labels, yseq[t].c_str());
147  }
148 
149  _inst.group = group;
150 
151  // Append the instance to the training set.
152  crfsuite_data_append(data, &_inst);
153 
154  // Finish the instance.
155  crfsuite_instance_finish(&_inst);
156 }

196 inline int _read_data(FILE *fpi, crfsuite_data_t* data, int group) {
197  int n = 0;
198  int lid = -1;
199  crfsuite_instance_t inst;
200  crfsuite_item_t item;
201  crfsuite_attribute_t cont;
202  iwa_t* iwa = NULL;
203  crfsuite_dictionary_t *attrs = data->attrs;
204  crfsuite_dictionary_t *labels = data->labels;
205  const iwa_token_t *token = NULL;
206  long filesize = 0, begin = 0, offset = 0;
207  int prev = 0, current = 0;
208 
209  /* Initialize the instance.*/
210  crfsuite_instance_init(&inst);
211  inst.group = group;
212 
213  /* Obtain the file size. */
214  begin = ftell(fpi);
215  fseek(fpi, 0, SEEK_END);
216  filesize = ftell(fpi) - begin;
217  fseek(fpi, begin, SEEK_SET);
218 
219  prev = 0;
220 
221  iwa = iwa_reader(fpi);
222 
223  while (token = iwa_read(iwa), token != NULL) {
224  /* Progress report. */
225  offset = ftell(fpi);
226  current = (int)((offset - begin) * 100.0 / (double)filesize);
227 
228  switch (token->type) {
229  case IWA_BOI:
230  /* Initialize an item. */
231  lid = -1;
232  crfsuite_item_init(&item);
233  break;
234 
235  case IWA_EOI:
236 
237  /* Append the item to the instance. */
238  if (0 <= lid) {
239  crfsuite_instance_append(&inst, &item, lid);
240  }
241 
242  crfsuite_item_finish(&item);
243  break;
244 
245  case IWA_ITEM:
246  if (lid == -1) {
247  lid = labels->get(labels, token->attr);
248  } else {
249  crfsuite_attribute_init(&cont);
250  cont.aid = attrs->get(attrs, token->attr);
251 
252  if (token->value && *token->value) {
253  cont.value = atof(token->value);
254  } else {
255  cont.value = 1.0;
256  }
257 
258  crfsuite_item_append_attribute(&item, &cont);
259  }
260 
261  break;
262 
263  case IWA_NONE:
264  case IWA_EOF:
265  /* Put the training instance. */
266  crfsuite_data_append(data, &inst);
267  crfsuite_instance_finish(&inst);
268  inst.group = group;
269  ++n;
270  break;
271  }
272  }
273 
274  return n;
275 }

443 void Tagger::set(const ItemSequence& xseq) {
444  int ret;
445  StringList yseq;
446  crfsuite_instance_t _inst;
447  crfsuite_dictionary_t *attrs = NULL;
448 
449  if (model == NULL || tagger == NULL) {
450  throw std::invalid_argument("The tagger is not opened");
451  }
452 
453  // Obtain the dictionary interface representing the attributes in the model.
454  if ((ret = model->get_attrs(model, &attrs))) {
455  throw std::runtime_error("Failed to obtain the dictionary interface for attributes");
456  }
457 
458  // Build an instance.
459  crfsuite_instance_init_n(&_inst, xseq.size());
460 
461  for (size_t t = 0; t < xseq.size(); ++t) {
462  const Item& item = xseq[t];
463  crfsuite_item_t* _item = &_inst.items[t];
464 
465  // Set the attributes in the item.
466  crfsuite_item_init(_item);
467 
468  for (size_t i = 0; i < item.size(); ++i) {
469  int aid = attrs->to_id(attrs, item[i].attr.c_str());
470 
471  if (0 <= aid) {
472  crfsuite_attribute_t cont;
473  crfsuite_attribute_set(&cont, aid, item[i].value);
474  crfsuite_item_append_attribute(_item, &cont);
475  }
476  }
477  }
478 
479  // Set the instance to the tagger.
480  if ((ret = tagger->set(tagger, &_inst))) {
481  crfsuite_instance_finish(&_inst);
482  attrs->release(attrs);
483  throw std::runtime_error("Failed to set the instance to the tagger.");
484  }
485 
486  crfsuite_instance_finish(&_inst);
487  attrs->release(attrs);
488 }

490 StringList Tagger::viterbi() {
491  int ret;
492  StringList yseq;
493  crfsuite_dictionary_t *labels = NULL;
494 
495  if (model == NULL || tagger == NULL) {
496  throw std::invalid_argument("The tagger is not opened");
497  }
498 
499  // Make sure that the current instance is not empty.
500  const size_t T = (size_t)tagger->length(tagger);
501 
502  if (T <= 0) {
503  return yseq;
504  }
505 
506  // Obtain the dictionary interface representing the labels in the model.
507  if ((ret = model->get_labels(model, &labels))) {
508  throw std::runtime_error("Failed to obtain the dictionary interface for labels");
509  }
510 
511  // Run the Viterbi algorithm.
512  floatval_t score;
513  int *path = new int[T];
514 
515  if ((ret = tagger->viterbi(tagger, path, &score))) {
516  delete[] path;
517  labels->release(labels);
518  throw std::runtime_error("Failed to find the Viterbi path.");
519  }
520 
521  // Convert the Viterbi path to a label sequence.
522  yseq.resize(T);
523 
524  for (size_t t = 0; t < T; ++t) {
525  const char *label = NULL;
526 
527  if (labels->to_string(labels, path[t], &label) != 0) {
528  delete[] path;
529  labels->release(labels);
530  throw std::runtime_error("Failed to convert a label identifier to string.");
531  }
532 
533  yseq[t] = label;
534  labels->free(labels, label);
535  }
536 
537  labels->release(labels);
538  return yseq;
539 }

541 double Tagger::probability(const StringList& yseq) {
542  int ret;
543  size_t T;
544  int *path = NULL;
545  std::stringstream msg;
546  floatval_t score, lognorm;
547  crfsuite_dictionary_t *labels = NULL;
548 
549  if (model == NULL || tagger == NULL) {
550  msg << "The tagger is not opened";
551  throw std::invalid_argument(msg.str());
552  }
553 
554  // Make sure that the current instance is not empty.
555  T = (size_t)tagger->length(tagger);
556 
557  if (T <= 0) {
558  return 0.;
559  }
560 
561  // Make sure that |y| == |x|.
562  if (yseq.size() != T) {
563  msg << "The numbers of items and labels differ: |x| = " << T << ", |y| = " << yseq.size();
564  throw std::invalid_argument(msg.str());
565  }
566 
567  // Obtain the dictionary interface representing the labels in the model.
568  if ((ret = model->get_labels(model, &labels))) {
569  msg << "Failed to obtain the dictionary interface for labels";
570  goto error_exit;
571  }
572 
573  // Convert string labels into label IDs.
574  path = new int[T];
575 
576  for (size_t t = 0; t < T; ++t) {
577  int l = labels->to_id(labels, yseq[t].c_str());
578 
579  if (l < 0) {
580  msg << "Failed to convert into label identifier: " << yseq[t];
581  goto error_exit;
582  }
583 
584  path[t] = l;
585  }
586 
587  // Compute the score of the path.
588  if ((ret = tagger->score(tagger, path, &score))) {
589  msg << "Failed to score the label sequence";
590  goto error_exit;
591  }
592 
593  // Compute the partition factor.
594  if ((ret = tagger->lognorm(tagger, &lognorm))) {
595  msg << "Failed to compute the partition factor";
596  goto error_exit;
597  }
598 
599  labels->release(labels);
600  delete[] path;
601  return std::exp((double)(score - lognorm));
602 
603 error_exit:
604 
605  if (labels != NULL) {
606  labels->release(labels);
607  labels = NULL;
608  }
609 
610  delete[] path;
611  throw std::runtime_error(msg.str());
612 }

614 double Tagger::marginal(const std::string& y, const int t) {
615  int l, ret, T;
616  floatval_t prob;
617  std::stringstream msg;
618  crfsuite_dictionary_t *labels = NULL;
619 
620  if (model == NULL || tagger == NULL) {
621  msg << "The tagger is not opened";
622  throw std::invalid_argument(msg.str());
623  }
624 
625  // Make sure that the current instance is not empty.
626  T = tagger->length(tagger);
627 
628  if (T <= 0) {
629  return 0.;
630  }
631 
632  // Make sure that 0 <= t < |x|.
633  if (t < 0 || T <= t) {
634  msg << "The position, " << t << "is out of range of " << T;
635  throw std::invalid_argument(msg.str());
636  }
637 
638  // Obtain the dictionary interface representing the labels in the model.
639  if ((ret = model->get_labels(model, &labels))) {
640  msg << "Failed to obtain the dictionary interface for labels";
641  goto error_exit;
642  }
643 
644  // Convert string labels into label IDs.
645  l = labels->to_id(labels, y.c_str());
646 
647  if (l < 0) {
648  msg << "Failed to convert into label identifier: " << y;
649  goto error_exit;
650  }
651 
652  // Compute the score of the path.
653  if ((ret = tagger->marginal_point(tagger, l, t, &prob))) {
654  msg << "Failed to compute the marginal probability of '" << y << "' at " << t;
655  goto error_exit;
656  }
657 
658  labels->release(labels);
659  return prob;
660 
661 error_exit:
662 
663  if (labels != NULL) {
664  labels->release(labels);
665  labels = NULL;
666  }
667 
668  throw std::runtime_error(msg.str());
669 }
